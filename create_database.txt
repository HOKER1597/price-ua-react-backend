-- Підключення до бази даних
\c cosmetick_ua;

-- Таблиця категорій
CREATE TABLE categories (
    id SERIAL PRIMARY KEY,
    name_en VARCHAR(100) NOT NULL UNIQUE,
    name_ua VARCHAR(100) NOT NULL
);

-- Таблиця брендів
CREATE TABLE brands (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE
);

-- Таблиця товарів
CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    category_id INTEGER NOT NULL REFERENCES categories(id),
    brand_id INTEGER NOT NULL REFERENCES brands(id),
    name VARCHAR(255) NOT NULL,
    volume VARCHAR(50),
    type VARCHAR(100),
    rating DECIMAL(3, 1),
    views INTEGER DEFAULT 0,
    code VARCHAR(50)
);

-- Таблиця додаткових зображень товарів
CREATE TABLE product_images (
    id SERIAL PRIMARY KEY,
    product_id INTEGER NOT NULL REFERENCES products(id),
    image_url VARCHAR(255) NOT NULL
);

-- Таблиця магазинів
CREATE TABLE stores (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    logo VARCHAR(255),
    years_with_us INTEGER,
    link VARCHAR(255)
);

-- Таблиця цін у магазинах
CREATE TABLE store_prices (
    id SERIAL PRIMARY KEY,
    product_id INTEGER NOT NULL REFERENCES products(id),
    store_id INTEGER NOT NULL REFERENCES stores(id),
    price DECIMAL(10, 2) NOT NULL,
    link VARCHAR(255),
    UNIQUE(product_id, store_id)
);

-- Таблиця деталей товарів
CREATE TABLE product_details (
    id SERIAL PRIMARY KEY,
    product_id INTEGER NOT NULL REFERENCES products(id),
    description TEXT, -- Короткий опис із products.description
    composition TEXT, -- Склад із products.composition
    usage TEXT, -- Інструкція з використання із products.usage
    description_full TEXT -- Опис із products.features.description
);

-- Таблиця характеристик товарів (без поля features)
CREATE TABLE product_features (
    id SERIAL PRIMARY KEY,
    product_id INTEGER NOT NULL REFERENCES products(id),
    brand VARCHAR(100),
    country VARCHAR(100),
    type VARCHAR(100),
    class VARCHAR(100),
    category VARCHAR(100),
    purpose VARCHAR(100), -- Із products.features.purpose
    gender VARCHAR(50),
    active_ingredients TEXT -- Із products.features.activeIngredients
);

-- Індекси для оптимізації пошуку
CREATE INDEX idx_products_category_id ON products(category_id);
CREATE INDEX idx_products_brand_id ON products(brand_id);
CREATE INDEX idx_product_images_product_id ON product_images(product_id);
CREATE INDEX idx_store_prices_product_id ON store_prices(product_id);
CREATE INDEX idx_store_prices_store_id ON store_prices(store_id);
CREATE INDEX idx_product_details_product_id ON product_details(product_id);
CREATE INDEX idx_product_features_product_id ON product_features(product_id);
